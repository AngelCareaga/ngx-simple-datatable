Template Code Examples

1. User Info Template
<ng-template ngxSimpleDatatableSlot="userInfo" let-data="data">
  <div class="flex items-center space-x-3">
    <img [src]="data.avatar" [alt]="data.name" class="w-12 h-12 rounded-full" />
    <div>
      <p class="font-semibold">{{ "{{ data.name }}" }}</p>
      <p class="text-sm text-gray-500">{{ "{{ data.email }}" }}</p>
    </div>
  </div>
</ng-template>
2. Progress Bar Template
<ng-template ngxSimpleDatatableSlot="progress" let-data="data">
  <div class="w-full">
    <div class="flex justify-between mb-1">
      <span>{{ "{{ data.progress }}" }}%</span>
    </div>
    <div class="w-full bg-gray-200 rounded-full h-2">
      <div class="bg-blue-500 h-2 rounded-full" [style.width.%]="data.progress"></div>
    </div>
  </div>
</ng-template>
